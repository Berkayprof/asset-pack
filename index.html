function setTab(btn) {
  const siblings = btn.closest('.tabs').querySelectorAll('.tab-btn');
  siblings.forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
}

function openModal() {
  document.getElementById('mijnModal').classList.add('open');
}

function closeModal(e) {
  // Sluit alleen als op de backdrop geklikt wordt
  if (e.target === e.currentTarget)
    e.currentTarget.classList.remove('open');
}
